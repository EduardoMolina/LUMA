203  template <typename NumType>
204  static void global_to_local(int i, int j, int k, GridObj* g, std::vector<NumType>& locals) {
205 
206  // Clear array if not empty
207  if (locals.size()) locals.clear();
208 
209  // Find indices in arrays
210  auto loc_i = std::find(g->XInd.begin(), g->XInd.end(), i);
211  auto loc_j = std::find(g->YInd.begin(), g->YInd.end(), j);
212  auto loc_k = std::find(g->ZInd.begin(), g->ZInd.end(), k);
213 
214  // Put indices in locals
215  if (loc_i != g->XInd.end()) locals.push_back( static_cast<NumType>(std::distance(g->XInd.begin(),loc_i)) );
216  else locals.push_back(-1);
217  if (loc_j != g->YInd.end()) locals.push_back( static_cast<NumType>(std::distance(g->YInd.begin(),loc_j)) );
218  else locals.push_back(-1);
219  if (loc_k != g->ZInd.end()) locals.push_back( static_cast<NumType>(std::distance(g->ZInd.begin(),loc_k)) );
220  else locals.push_back(-1);
221 
222  return; // Directions not on the grid are returned with index -1
223

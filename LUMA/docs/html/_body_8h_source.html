183 template <typename MarkerType>
184 void Body<MarkerType>::passToVoxelFilter(double x, double y, double z, int& curr_mark, std::vector<int>& counter) {
185 
186  // If point in current voxel
187  if (isInVoxel(x, y, z, curr_mark)) {
188 
189  // Increment point counter
190  counter[curr_mark]++;
191 
192  // Update position of marker in current voxel
193  markers[curr_mark].position[0] =
194  ((markers[curr_mark].position[0] * (counter[curr_mark] - 1)) + x) / counter[curr_mark];
195  markers[curr_mark].position[1] =
196  ((markers[curr_mark].position[1] * (counter[curr_mark] - 1)) + y) / counter[curr_mark];
197  markers[curr_mark].position[2] =
198  ((markers[curr_mark].position[2] * (counter[curr_mark] - 1)) + z) / counter[curr_mark];
199 
200 
201  // If point is in an existing voxel
202  }
203  else if (isVoxelMarkerVoxel(x, y, z)) {
204 
205  // Recover voxel number
206  MarkerData* m_data = getMarkerData(x, y, z);
207  curr_mark = m_data->ID;
208 
209  // Increment point counter
210  counter[curr_mark]++;
211 
212  // Update position of marker in current voxel
213  markers[curr_mark].position[0] =
214  ((markers[curr_mark].position[0] * (counter[curr_mark] - 1)) + x) / counter[curr_mark];
215  markers[curr_mark].position[1] =
216  ((markers[curr_mark].position[1] * (counter[curr_mark] - 1)) + y) / counter[curr_mark];
217  markers[curr_mark].position[2] =
218  ((markers[curr_mark].position[2] * (counter[curr_mark] - 1)) + z) / counter[curr_mark];
219 
220  delete m_data;
221 
222  }
223  // Must be in a new marker voxel
224  else {
225 
226  // Reset counter and increment voxel index
227  curr_mark = static_cast<int>(counter.size());
228  counter.push_back(1);
229 
230  // Create new marker as this is a new marker voxel
231  addMarker(x, y, z);

248 template <typename MarkerType>
249 bool Body<MarkerType>::isInVoxel(double x, double y, double z, int curr_mark) {
250 
251  try {
252 
253  // Try to retrieve the position of the voxel centre belonging to <curr_mark>
254  // Assume that first support point is the closest to the marker position
255  double vx = _Owner->XPos[markers[curr_mark].supp_i[0]];
256  double vy = _Owner->YPos[markers[curr_mark].supp_j[0]];
257  double vz = _Owner->ZPos[markers[curr_mark].supp_k[0]];
258 
259  // Test within
260  if ((x >= vx - (_Owner->dh / 2) && x < vx + (_Owner->dh / 2)) &&
261  (y >= vy - (_Owner->dh / 2) && y < vy + (_Owner->dh / 2)) &&
262  (z >= vz - (_Owner->dh / 2) && z < vz + (_Owner->dh / 2))
263  ) return true;
264 
265  // Catch all
266  }
267  catch (...) {
268 
269  // If failed, marker probably doesn't exist so return false
270  return false;
271 
272  }
273 
274  return false;
